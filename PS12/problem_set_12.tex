\problem[(3 pt)]{}

Let $X$ ba a $k$-scheme. Describe a natural bijection between $\on{Mor}_k(\on{Spec}k[\epsilon]/(\epsilon^2),X)$ and the set of $k$-points $p$ of $X$ with the additional datum of a tangent vector at $p$. In particular, the subset of morphisms mapping $\on{Spec}(k)=\on{Spec}(k[\epsilon]/(\epsilon^2))_{red}$ to $p$ admits the structure of a $k$-vector space. Describe the differential of a morphism $f\colon X\to Y$ at $p$ in these terms when $k(p)=k(f(p))$.

\problem[(3 pt)]{}
Let $X$ be a $k$-scheme of finite type, $D\subseteq X$ an effective Cartier divisor (a hypersurface which is locally cut out by a non-zero divisor in $X$). If $D$ is regular at a point $p$, then so is $X$.

Is this still true if $D$ is the vanishing locus of a zero divisor?

\problem[(3 pt)]{}
Let $C=V(y^2-x^3)\subseteq\Aaff^2_k$ with $k=\bar k$ and $\on{char}(k)\neq 2,3$ be an ``ordinary cusp''.
\begin{enumerate}
 \item Compute the singular locus of $C$.
 \item Let $\tilde{A}$ denote the blow-up of $\Aaff^2$ in $C^{sing}$, and let $C'$ denote the strict transform of $C$. Compute the intersection of $C'$ with the exceptional divisor in $\tilde{A}$, and the tangent cone of $C$ at the origin of $\Aaff^2$.
 \item Show that $C$ is not isomorphic to $\Aaff^1$, but $C'$ is, and show that the morphism $C'\to C$ is finite and birational.
\end{enumerate}


\problem[(3 pt)]{}
Let us work over $k=\bar k$ with $\on{char}(k)\neq 2,3$.
\begin{enumerate}
 \item Let $X$ be the cone over the twisted cubic. Compute $X^{sing}$. What is the exceptional divisor of the blow-up of $X$ at $X^{sing}$?
 \item Let $Y$ be the blow-up of $\Aaff^3$ along the union of two intersecting lines $V(xy,z)$. Compute $Y^{sing}$. Show that $Y$ is not isomorphic to what one gets by blowing up first one line, and then the strict transform of the other.
 \item Let $Z$ be the surface $V(x^2-yz^2)$ in $\Aaff^3$ (this ``pinch point'' is a higher-dimensional analogue of the singularity encountered in the previous problem). Compute $Z^{sing}$. Compute the blow-up of $Z$ at the origin.
\end{enumerate}
