\problem[(3 pt)]{}

Linear subspaces and projectivities.

\begin{enumerate}
 \item Let $r$ and $s$ be disjoint lines in $\PP^3(k)$. For every point $p\in\PP^3(k)\setminus(r\cup s)$, there exists a unique line $l_p$ passing through $p$ and intersecting both $r$ and $s$.
 \item Let $\phi$ be a projectivity of $\PP^n(k)$. Show that the locus of fixed points of $\phi$ is a union of linear supspaces $L_1,\ldots,L_r$, such that $L_i$ does not meet the linear span of $(L_j)_{j\neq i}$.
 \item Suppose that the fixed locus of $\phi\in PGL_4(k)$ consists of two lines $r$ and $s$. For every point $p\in\PP^3(k)\setminus(r\cup s)$, the line joining $p$ to $f(p)$ meets both $r$ and $s$.
\end{enumerate}

\problem[(2 pt)]{}
Let $r$ be a line in $\PP^2(\CC)$. Show that $r$ contains either one or infinitely many real points. (Hint: use conjugation.)

\problem[(3 pt)]{}
Let $k$ be algebraically closed. Let $p_1,\ldots,p_5$ be distinct points of $\PP^2(k)$.
\begin{enumerate}
 \item Show that there exists at least a conic $C$ passing through $p_1,\ldots,p_5$.
 \item $C$ is non-degenerate (reduced and irreducible) if and only if any three of the points are not collinear. Moreover, in this case $C$ is unique.
 \item If $C$ is degenerate, then it is unique if and only if any four of the points are not collinear.
\end{enumerate}
