\problem[(3 pt)]{}
Show that the following properties satisfy the assumptions of the Affine Communication Lemma [Vakil, Prop. 5.3.2].
\begin{enumerate}
 \item locally of finite type over a base ring $R$;
 \item reduced.
\end{enumerate}

\problem[(2 pt)]{}
Let $X=V(x_0x_2-x_1^2,x_0x_3-x_1x_2,x_1x_3-x_2^2)\subseteq \mathbb P^3$. Show that $X$ is isomorphic to $\PP^1$. Show that no two of the three equations above are enough to cut out $X$.

\problem[(3 pt)]{}
Let $R=\bigoplus R_d$ be a $\ZZ$-graded ring. Prove the following facts.
\begin{enumerate}
 \item A homogeneous ideal $\mathfrak p\subsetneq R$ is prime if and only if, for any pair of \emph{homogeneous} elements $a,b$ in $R$, $ab\in\mathfrak p$ implies $a\in\mathfrak p$ or $b\in\mathfrak p$.
 \item Let $S\subseteq R$ be a multiplicative set consisting of homogeneous elements. Then $S^{-1}R$ admits a structure of $\ZZ$-graded ring making $R\to S^{-1}R$ into a graded homomorphism (of degree 0).
 \item Every maximal homogeneous ideal of $R$ is of the form:
 \[\ldots\oplus R_{-1}\oplus\mathfrak m\oplus R_1\oplus\ldots\]
 for some maximal ideal $\mathfrak m$ of $R_0$.
\end{enumerate}

\problem[(2 pt)]{}
Let $R=\bigoplus R_d$ be a $\ZZ$-graded ring, and $f\in R$ an invertible homogeneous element of positive degree. Then there is a bijection between prime ideals of $R_0$ and homogeneous prime ideals of $R$. (Hint: given $\mathfrak p\in \on{Spec}(R_0)$, define $P=\bigoplus P_d$ where $g\in P_d$ if and only if $\frac{g^{\deg(f)}}{f^d}\in\mathfrak p$. Show that $P$ is a homogeneous prime ideal of $R$, and that $P\cap R_0=\mathfrak p$.)
