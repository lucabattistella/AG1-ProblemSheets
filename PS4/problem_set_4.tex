\problem[(1 pt)]{}
Show that every affine scheme with the Zariski topology is quasi-compact.

\problem[(2 pt)]{}
\begin{enumerate}
 \item Show that every scheme admits a unique morphism to $\Spec(\ZZ)$.
 \item Show that global regular functions on a scheme $X$ are in bijection with morphisms \[X\to\Aaff^1_{\ZZ}=\Spec(\ZZ[x]).\]
\item Consider the ring homomorphisms $\alpha\colon \ZZ[x]\to\ZZ[x]\otimes_{\ZZ}\ZZ[x]$ mapping $x$ to $x\otimes1+1\otimes x$, and $\mu\colon \ZZ[x]\to\ZZ[x]\otimes_{\ZZ}\ZZ[x]$ mapping $x$ to $x\otimes x$. Can you interpret them in view of the previous point?
\end{enumerate}

\problem[(2 pt)]{}
\begin{enumerate}
 \item Consider the scheme $U=\mathbb{A}^2_{\ZZ}\setminus V(x_1,x_2)$. Show that it is not affine. (Hint: compute global regular functions.)
 \item Produce an example of a scheme with an affine cover, such that the intersection of two affine opens is not affine. (Hint: gluing.)
\end{enumerate}

\problem[(3 pt)]{}
Let $(f,f^\#):X\to Y$ be a morphism of affine algebraic varieties over $\kk$. Prove or disprove:
\begin{enumerate}
 \item $f$ is surjective if and only if $f^\#$ is injective.
 \item $f$ is injective if and only if $f^\#$ is surjective.
\end{enumerate}
Would anything be different if we considered a morphism of affine schemes?

\problem[(3 pt)]{}
\begin{enumerate}
 \item Show that $\Spec(\CC\otimes_{\RR}\CC)$ consists of two points.
 \item Let $X$ and $Y$ be affine algebraic varieties over an algebraically closed field $\kk$. Show that $X\times Y$ is an affine algebraic variety, whose underlying set $\underline{X\times Y}$ is $\underline{X}\times\underline{Y}$.
 \item Show that if $X$ and $Y$ are irreducible, then so is $X\times Y$.
\end{enumerate}
